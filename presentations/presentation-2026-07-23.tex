% ---------------------------------------------------------------- HOJA DE RUTA
\begin{frame}{La pregunta, y el argumento en una lámina}
\begin{itemize}
  \item ¿Qué organiza la colaboración política cuando \emphr{no hay historia acumulada} que la organice?
  \vspace{0.25cm}
  \item La Convención como laboratorio: mayoría novata, órgano que se disuelve, reglas propias.
  \vspace{0.25cm}
  \item El argumento en tres actos:
  \begin{enumerate}
    \item \emphr{Lo que la gente traía puesto} formó la red: territorio, lista, ideología — \emph{no} las credenciales.
    \item El capital relacional \emphr{se acumula}: la Convención se tejió \emph{de a dos}.
    \item Qué compra ese capital: \emphr{mentes no} $\cdot$ \emphr{conducta sí} $\cdot$ \emphr{textos sí}.
  \end{enumerate}
  \vspace{0.25cm}
  \item Cierre: \emphr{la aritmética del 103} — con las mentes quietas, la geometría era destino.
\end{itemize}
\end{frame}

\sectionslide{Acto I}{Lo que la gente traía puesto}

\begin{frame}{El modelo de elección de firma (logit condicional)}
\begin{itemize}
  \item Cada iniciativa $a$ es un \emphr{menú}: los 154 deciden firmar o no. Utilidad latente:
\end{itemize}
$$U_{ia} = \beta^\top x_{ia} + \alpha_a + \varepsilon_{ia}
\qquad\Rightarrow\qquad
P(S_a \mid |S_a|) = \frac{\exp(\sum_{i \in S_a}\beta^\top x_{ia})}{\sum_{|R|=|S_a|}\exp(\sum_{i \in R}\beta^\top x_{ia})}$$
\begin{itemize}
  \item El atractivo del texto ($\alpha_a$: tema, redactor, momento) \emphr{se cancela}: solo comparaciones \emph{dentro} del mismo menú.
  \item 947 menús $\times$ 154 = 145.838 decisiones; covariables leave-one-out respecto de la coalición; EE cluster por convencional.
\end{itemize}
\end{frame}

\sectionslide{Acto II}{El capital relacional se acumula}

\begin{frame}{El modelo de eventos (RHEM): la pregunta que la foto no responde}
\begin{itemize}
  \item Las 947 iniciativas ocurren \emphr{en orden} (100\% fechadas). Cuando dos firman juntos, ¿es porque \emph{se parecen} o porque \emph{ya habían firmado juntos}?
  \item Cada iniciativa = una escena: la coalición real contra 50 coaliciones ficticias del mismo tamaño ese mismo día.
  \item La estadística de historia clave — \emphr{familiaridad de pares}:
\end{itemize}
$$sub.rep^{(2)}(t,h) = \binom{|h|}{2}^{-1} \sum_{\{i,j\} \subseteq h} \#\{\text{eventos previos a } t \text{ que contienen a } i \text{ y } j\}$$
\begin{itemize}
  \item \nota{En dos memorias: infinita y semivida de 15 días. Mismas covariables de composición que el clogit, para leer las tablas en paralelo.}
\end{itemize}
\end{frame}

\sectionslide{Acto III}{¿Qué compra el capital relacional?}

\begin{frame}{Pregunta 1: ¿compra mentes? El diseño}
\begin{itemize}
  \item Exposición de $i$ en la onda $t$: posición media de sus co-firmantes, ponderada por intensidad:
\end{itemize}
$$E_{i,t} = \frac{\sum_{j \neq i} w_{ij,t}\,\theta_{j,t}}{\sum_{j \neq i} w_{ij,t}}
\qquad\qquad
\Delta\theta_{i,t} = \alpha_i + \beta\,\theta_{i,t-1} + \lambda\,E_{i,t-1} + \varepsilon_{it}$$
\begin{itemize}
  \item Efectos fijos individuales (solo variación \emph{dentro} de cada persona), EE cluster; 4.355 obs.
  \item Resultado: $\hat\lambda = +0.007$ (EE $0.004$), \emphr{nulo}. Robusto a 3 ventanas temporales $\times$ 7 dosis de exposición.
\end{itemize}
\end{frame}

\begin{frame}{Selección, no influencia — en una imagen}
\fig{0.88}{m2_selection_vs_influence_preview.pdf}
\vspace{0.05cm}
\nota{(a) Entre personas: posición y vecindario van pegados ($r=0.95$) — elegimos vecindarios que se nos parecen. (b) Dentro de cada persona, los cambios no se siguen. Ámbar: el 5\% más desalineado — los únicos con espacio para ser arrastrados — cae en (b) sin pendiente ($r=-0.04$).}
\end{frame}

\begin{frame}{Cuatro defensas del nulo}
\begin{enumerate}
  \item \emphr{Falsificación fina}: exposición pasada y futura están pegadas ($r=0.995$); el test limpio separa la \emph{novedad} del futuro — y es la novedad la que predice el cambio de hoy ($+0.06$, $p=.007$), con el pasado en cero. \nota{Me muevo hacia donde \emph{estará} mi vecindario: definición operativa de selección.}
  \item \emphr{Espacio}: la selección cerró 74\% de la distancia — pero el 0.59 restante era espacio real donde la influencia se habría visto.
  \item \emphr{Instrumento}: efecto mínimo detectable $\lambda = 0.011$; lo estimado ($0.007$) queda debajo.
  \item \emphr{Error de medición propagado}: bootstrap paramétrico del dynIRT (el ruido de $\theta$ es 3.6$\times$ el movimiento corto) $\to$ el EE de $\lambda$ apenas se mueve ($0.0036 \to 0.0039$): el nulo \emphr{no es artefacto del termómetro}.
\end{enumerate}
\end{frame}

\begin{frame}{Síntesis}
\begin{itemize}
  \item En una asamblea de extraños, la colaboración se organizó con lo que la gente \emphr{traía puesto}: territorio, lista, ideología — no credenciales.
  \vspace{0.2cm}
  \item Sobre esos primeros encuentros, la red se rigidizó rápido: \emphr{se tejió de a dos}, y la afinidad conservó fuerza propia.
  \vspace{0.2cm}
  \item Ese capital \emphr{no compró mentes} (selección, con el nulo defendido cuatro veces), pero \emphr{sí conducta} (la defección viaja por los lazos, transportada por los novatos) y \emphr{sí textos} (equipos con historia, cerca del pívot, anchos).
  \vspace{0.2cm}
  \item Y la institución cierra el círculo: la regla de 2/3 fijó un pívot, y la red fue el mecanismo por el que esa aritmética se volvió comportamiento.
